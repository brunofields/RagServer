aegis_guild,27,19,4	script	Skill Master	654,{
	set @eac, eaclass();

	mes .npc_name$;
	mes "Hello! I can instantly grant you all skills available for your current class.";
	mes "Please note, my service fee depends on your first job's level when you changed class.";
	next;
	mes .npc_name$;
	mes "^1A66FFJob Level 50^000000: ^FF3355" + callfunc("F_InsertComma", .BaseFee) + "^000000 ^777777Zeny^000000";
	mes "^1A66FFBelow Job Level 50^000000: ^FF3355" + callfunc("F_InsertComma", .BaseFee * 2) + "^000000 ^777777Zeny^000000";
	mes "^1A66FFNote^000000: ^FF3355Super Novice^000000, ^FF3355TaeKwon Kid^000000, ^FF3355Gunslinger^000000, and ^FF3355Ninja^000000 are always charged the ^1A66FF50 job level^000000 rate.^000000";
	next;

	switch(select("^1A66FFYes^000000:What is this?:^777777No^000000")) {
		case 1:
			if ((@eac&EAJ_UPPERMASK) == EAJ_SUPER_NOVICE) {
				.@maxjob = 99;
			} else if ((@eac&EAJL_UPPER && @eac&EAJL_2) || BaseClass == Job_Gunslinger || BaseClass == Job_Ninja) {
				.@maxjob = 70;
			} else if ((@eac&EAJL_2) || BaseClass == Job_Taekwon) {
				.@maxjob = 50;
			} else {
				mes "Come back to me when you are stronger.";
				close;
			}

			if (BaseLevel < 99 || JobLevel < .@maxjob) {
				mes "Come back to me when your Base Level ^1A66FF99^000000 and Job Level ^1A66FF" + .@maxjob + "^000000.";
				close;
			}

			.@JobFeeMod = (jobchange_level > 10 && jobchange_level < 50) ? 2 : 1;
			.@FinalFee = .BaseFee * .@JobFeeMod;

			if (!skillmastery && Zeny < .@FinalFee) {
				mes "You don't have enough zeny.";
				close;
			}

			if (skillmastery == 2 || (skillmastery == 1 && jobchange_level == 50)) {
				// Already granted full skill mastery
				specialeffect2 EF_ANGEL;
				atcommand "@skillall " + strcharinfo(0);
				mes .npc_name$;
				mes "You have already used my service and met the required job level.";
				mes "There is no need to pay again.";
				close;
			}

			if (skillmastery) {
				// Re-purchase upgrade
				if (Zeny < .BaseFee) {
					mes "You don't have enough zeny.";
					close;
				}
				skillmastery = 2;
				Zeny -= .BaseFee;
			} else {
				// First-time purchase
				if (jobchange_level == 10) {
					skillmastery = 2;
				} else {
					skillmastery = .@JobFeeMod;
				}
				Zeny -= .@FinalFee;
			}

			specialeffect2 EF_ANGEL;
			atcommand "@skillall " + strcharinfo(0);
			mes .npc_name$;
			mes "All done!";
			mes "You've now mastered all the skills available to your class.";
			close;

		case 2:
			mes .npc_name$;
			mes "With my service, you can instantly learn all the skills available to your current class.";
			next;
			switch(select("^1A66FFView Eligibility Requirements^000000:^777777Thanks for the info!^000000")) {
				case 1:
					mes .npc_name$;
					mes "To use my service, you must meet the following base/job level conditions:";
					mes "^00A000Second Class (Non-Transcendent)^000000: ^1A66FF99/50^000000";
					mes "^00A000Second Class (Transcendent)^000000: ^1A66FF99/70^000000";
					mes "^00A000Gunslinger / Ninja^000000: ^1A66FF99/70^000000";
					mes "^00A000TaeKwon, TaeKwon Master, Soul Linker^000000: ^1A66FF99/50^000000";
					mes "^00A000Super Novice^000000: ^1A66FF99/99^000000";
					close;
				case 2:
					mes .npc_name$;
					mes "You're welcome. Come back anytime.";
					close;
			}
		case 3:
			mes .npc_name$;
			mes "Okay then.";
			close;
	}
	close;

OnInit:
	.npc_name$ = "[Skill Master]";
	.BaseFee = 10000000;	// Fee (In Zeny)
}
